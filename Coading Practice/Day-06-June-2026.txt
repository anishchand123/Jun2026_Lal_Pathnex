## Debugging Pipeline

stages:
  - build
  - test

build:
  stage: build
  image: maven:3.8.1-jdk-17
  script:
    - git clone https://github.com/Pathnex/sample-java-app.git
    - cd sample-java-app
    - mvn clean compile
  artifacts:
    paths:
      - sample-java-app/

test:
  stage: test
  image: maven:3.8.1-jdk-17
  dependencies:
    - build
  script:
    - cd sample-java-app
    - mvn test
