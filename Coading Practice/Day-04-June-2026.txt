## Terraform - EC2 with key pair (c6i.8xlarge)

provider "aws" {
    rergion = "us-east-1"
}

resource "aws_key_pair" "Pathnexkey" {
    key_name = "PathnexKey"
    Public_Key = "ssh-rsa AAAA...."
}

resource "aws_instance" "PathnexEC2" {
    ami     =   "ami-0abcd1234abcd1234"
    instance_type = "c6i.8xlarge"
    key_name    =   aws_key_pair.PathnexKey.key_name

    tags = {
    Name = "Pathnex-EC2"

## Jenkins Pipeline - Parallel Build & Test
You will learn how to **run build and test in parallel8*

pipeline {
    agent any
    stages {
        stage('Parallel Tasks') {
            Parallel {
                stage('Build') {
                    steps {
                        sh 'mvn clean cmpile'
                    }
                }
                stage('Test') {
                    steps {
                        sh 'mvn test'
                    }
                }          
            }
        }
    }
}

GitLab CI - Parallel Jobs
You will learn how to **run parallel jobs in GitLab CI**

Stages:
    - build-test

maven-build:
    stage: build-test
    image: maven:3.8.1-jdk-17
    script:
        - git clone https://github.com/Pathnex/sample-java-app.git
        - sample-java-app
        - mvn clean cmpile
    parallel: 1

maven-test:
    stage: build-test
    image: maven:3.8.1-jdk-17
    script:
        - cd sample-java-app
        - mvn test
    parallel: 1
